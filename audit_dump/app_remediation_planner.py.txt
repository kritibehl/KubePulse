    30	        reasoning.append("Readiness stayed green despite degraded service quality")

    35

    56	        deployment_decision = "deploy_with_caution"

    58	    else:

    68	        "all_actions": deduped_actions,
    69	        "reasoning": reasoning,
    70	        "urgency": urgency,
    71	        "deployment_decision": deployment_decision,
    72	        "operator_goal": "restore safe-to-operate state" if not safe_to_operate else "maintain healthy rollout posture",
    73	    }
