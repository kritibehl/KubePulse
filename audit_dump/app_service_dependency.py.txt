     8	        {"source": target, "target": "shared-db", "protocol": "tcp"},
     9	    ]
    10	
    11	    latency_error_propagation_path = [edge["source"] for edge in edges] + [edges[-1]["target"]]

    14	    scenario = report.get("scenario", "unknown")

    18	    elif scenario in {"packet_loss", "tcp_resets", "connection_churn", "mtu_mismatch"}:

    20	        impacted_segment = "inter-service TCP path"
    21	    elif scenario in {"node_partition", "dropped_egress", "degraded_ingress"}:
    22	        likely_root_cause = "node or edge network segment"
    23	        impacted_segment = "cross-node / ingress-egress network path"
    24	    else:
    25	        likely_root_cause = target
    26	        impacted_segment = "upstream dependency path"
    27	
    28	    return {
    29	        "services": blast_radius_services,
    30	        "edges": edges,
    31	        "latency_error_propagation_path": latency_error_propagation_path,
    32	        "blast_radius_services": blast_radius_services,
    33	        "likely_root_cause": likely_root_cause,
    34	        "impacted_network_segment": impacted_segment,
    35	        "upstream_service": source,
    36	        "downstream_service": target,
    37	    }
