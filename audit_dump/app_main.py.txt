    58	def _finalize_result(result: dict) -> dict:
    59	    result.update(compute_resilience_score(result))
    60	    result.update(compute_network_health_score(result))
    61	    result.update(evaluate_slo(result))
    62	    if result.get("baseline_path") or result.get("final_path"):
    63	        result.update(correlate_path_trace(result))
    64	    scenario_spec = {}
    65	    scenario_file = Path("scenarios") / f"{result.get('scenario')}.yaml"
    66	    if scenario_file.exists():
    67	        import yaml
    68	        scenario_spec = yaml.safe_load(scenario_file.read_text()) or {}
    69	    result.update(build_compare_view(result))
    70	    result.update(build_probe_gap(result))
    71	    result.update(build_dependency_path_report(result))
    72	    result.update(evaluate_thresholds(result, scenario_spec))
    73	    result.update(build_resilience_explanation(result))
    74	    result["safe_to_operate"] = bool(
    75	        result.get("availability_ok", False)
    76	        and result.get("p95_ok", False)
    77	        and result.get("p99_ok", False)
    78	        and result.get("error_rate_ok", False)
    79	        and result.get("slo_met", False)
    80	        and not result.get("readiness_false_positive", False)
    81	    )
    82	    result["probes_say_healthy"] = bool(result.get("readiness_before") == "ready")
    83	    if not result.get("recommendation_action"):
    84	        result["recommendation_action"] = (
    85	            "continue" if result["safe_to_operate"]
    86	            else ("block" if result.get("status") == "fail" else "reroute")
    87	        )
    88	    result.update(build_decision_report(result))
    89	    dependency_analysis = infer_dependency_analysis(result)
    90	    result.update(dependency_analysis)
    91	    remediation = recommend_network_remediation(result, dependency_analysis)
    92	    result.update(remediation)
    93	    result["recommendation"] = f"{remediation['recommended_action']} | {remediation['suggested_config_change']}"
    94	    report_path = save_report(result)
    95	    result["report_path"] = report_path
    96	    run_id = persist_report(result)
    97	    result["run_id"] = run_id
    98	    rollout_risk = compute_rollout_risk(result)
    99	    remediation_plan = build_remediation_plan(result, rollout_risk)
   100	    operator_action_plan = build_operator_action_plan(result, rollout_risk, remediation_plan)
   101	
   102	    result["rollout_risk"] = rollout_risk
   103	    result["remediation_plan"] = remediation_plan
   104	    result["operator_action_plan"] = operator_action_plan
   105	    result["deployment_decision"] = remediation_plan.get("deployment_decision")
   106	
   107	    return result

   109	def _network_response(builder, request: NetworkScenarioRequest) -> ResilienceReport:
   110	    result = builder(
   111	        namespace=request.namespace,
   112	        source_service=request.source_service,
   113	        target_service=request.target_service,
   114	        dry_run=request.dry_run,
   115	        run_group=request.run_group,
   116	        run_kind=request.run_kind,
   117	    )
   118	    result = _finalize_result(result)
   119	    return ResilienceReport(**result)
